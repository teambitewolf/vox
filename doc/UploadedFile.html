def initialize(attrs)
  return if attrs.nil?

  @file_name = attrs.fetch :filename
  @file_ext  = @file_name.split('.').last
  @file_type = attrs.fetch :type
  @file_cat  = @file_type.split('/').first
  @temp_file = attrs.fetch :tempfile
end
